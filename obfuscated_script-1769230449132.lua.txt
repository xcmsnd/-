    local particleFrame = Instance.new("Frame", parent)
    particleFrame.Size = UDim2.new(1, 0, 1, 0)
    particleFrame.BackgroundTransparency = 1
    for i = 1, 25 do
        local particle = Instance.new("Frame", particleFrame)
        particle.Size = UDim2.new(0, math.random(2, 5), 0, math.random(2, 5))
        particle.BackgroundColor3 = Color3.fromRGB(100, 100, 255)
        particle.BackgroundTransparency = 0.3
        Instance.new("UICorner", particle).CornerRadius = UDim.new(1, 0)
        local xPos = math.random(-20, 100) / 100
        local yPos = math.random(-20, 100) / 100
        task.spawn(function()
            while particle and particle.Parent do
                particle.Position = UDim2.new(xPos, 0, yPos, 0)
                particle.BackgroundTransparency = 0.2 + math.abs(math.sin(tick() + i))
                xPos = (xPos + 0.001) % 1.2
                yPos = (yPos + 0.0015) % 1.2
                task.wait(0.05)
            end
        end)
    end
end
